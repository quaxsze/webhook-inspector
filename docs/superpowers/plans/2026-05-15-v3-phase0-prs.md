> **Convention secrets (à respecter sur tous les PRs qui touchent à des secrets — F1 HMAC en PR1, F5 forward en PR7) :**
>
> - **Entité domaine** porte le secret **chiffré** : `signature_secret_encrypted: bytes | None`, `forward_secret_encrypted: bytes | None`. Aucune référence à la clé maître dans le domaine.
> - **Repository** persiste/lit les bytes tels quels — pas de crypto dans la couche infrastructure repo.
> - **Use case** (qui injecte la `secrets_key`) chiffre le plaintext entrant via `encrypt_secret(key, plaintext)` avant assignation à l'entité, et déchiffre via `decrypt_secret(key, encrypted)` quand il a besoin du plaintext (validation HMAC, signature de forward).
> - **API Pydantic** parle de plaintext (`secret: str`) côté input ; jamais de plaintext en réponse, le secret n'est ni listé ni renvoyé une fois posé.
>
> Cette convention élimine l'ambiguïté entre `signature_secret` et `signature_secret_encrypted` qui traînait dans la version initiale du plan.

```python
"""AES-256-GCM helpers for encrypting at-rest secrets (HMAC secrets, Forward secrets).

Format of ciphertext blob: nonce (12 bytes) || ciphertext+tag (variable).
"""

import os

from cryptography.hazmat.primitives.ciphers.aead import AESGCM

_NONCE_SIZE = 12


def encrypt_secret(key: bytes, plaintext: str) -> bytes:
    if len(key) != 32:
        raise ValueError("key must be 32 bytes for AES-256-GCM")
    aesgcm = AESGCM(key)
    nonce = os.urandom(_NONCE_SIZE)
    ciphertext = aesgcm.encrypt(nonce, plaintext.encode("utf-8"), associated_data=None)
    return nonce + ciphertext


def decrypt_secret(key: bytes, blob: bytes) -> str:
    if len(key) != 32:
        raise ValueError("key must be 32 bytes for AES-256-GCM")
    if len(blob) < _NONCE_SIZE:
        raise ValueError("blob too short")
    aesgcm = AESGCM(key)
    nonce, ciphertext = blob[:_NONCE_SIZE], blob[_NONCE_SIZE:]
    return aesgcm.decrypt(nonce, ciphertext, associated_data=None).decode("utf-8")
```

### Step 9: Implement Stripe validator (TDD)

Stripe documents the signature scheme: header `Stripe-Signature: t=<timestamp>,v1=<sha256-hex>`. Computed signature = `HMAC_SHA256(secret, "<timestamp>.<payload>")`.

Create `tests/unit/domain/services/hmac/test_stripe.py`:
```python
import hmac
import hashlib
import time

from webhook_inspector.domain.services.hmac.stripe import StripeValidator
from webhook_inspector.domain.services.hmac.base import ValidationResult


def _make_stripe_signature(secret: str, payload: bytes, timestamp: int | None = None) -> str:
    ts = timestamp or int(time.time())
    signed = f"{ts}.{payload.decode('utf-8')}".encode()
    sig = hmac.new(secret.encode(), signed, hashlib.sha256).hexdigest()
    return f"t={ts},v1={sig}"


def test_stripe_valid_signature_returns_valid():
    secret = "whsec_testsecret"
    body = b'{"id":"evt_1","type":"payment_intent.succeeded"}'
    sig = _make_stripe_signature(secret, body)
    validator = StripeValidator()
    assert (
        validator.validate(body=body, headers={"stripe-signature": sig}, secret=secret)
        == ValidationResult.VALID
    )


def test_stripe_wrong_secret_returns_invalid():
    body = b'{"id":"evt_1"}'
    sig = _make_stripe_signature("whsec_one", body)
    validator = StripeValidator()
    assert (
        validator.validate(body=body, headers={"stripe-signature": sig}, secret="whsec_two")
        == ValidationResult.INVALID
    )


def test_stripe_missing_header_returns_missing():
    validator = StripeValidator()
    assert (
        validator.validate(body=b"x", headers={}, secret="whsec_anything")
        == ValidationResult.MISSING
    )


def test_stripe_tampered_body_returns_invalid():
    secret = "whsec_testsecret"
    body = b'{"id":"evt_1"}'
    sig = _make_stripe_signature(secret, body)
    validator = StripeValidator()
    assert (
        validator.validate(body=b'{"id":"evt_2"}', headers={"stripe-signature": sig}, secret=secret)
        == ValidationResult.INVALID
    )


def test_stripe_multiple_v1_signatures_accept_any_valid():
    # During key rotation, Stripe sends multiple v1= in the same header.
    # If at least ONE matches our secret, the request is valid.
    secret_current = "whsec_current"
    secret_old = "whsec_old"
    body = b'{"id":"evt_1"}'
    ts = "1700000000"
    import hmac as _hmac, hashlib as _hashlib
    signed = f"{ts}.{body.decode()}".encode()
    sig_current = _hmac.new(secret_current.encode(), signed, _hashlib.sha256).hexdigest()
    sig_old = _hmac.new(secret_old.encode(), signed, _hashlib.sha256).hexdigest()
    # Header has both signatures
    header = f"t={ts},v1={sig_old},v1={sig_current}"
    validator = StripeValidator()
    assert (
        validator.validate(body=body, headers={"stripe-signature": header}, secret=secret_current)
        == ValidationResult.VALID
    )
```

```python
import hashlib
import hmac

from webhook_inspector.domain.services.hmac.base import HmacValidator, ValidationResult


class StripeValidator(HmacValidator):
    """Stripe webhook signature validator.

    Reference: https://docs.stripe.com/webhooks#verify-manually
    Format: Stripe-Signature: t=<ts>,v1=<sha256_hex>[,v1=<another>]
    Signed payload: f"{ts}.{body}".encode()
    """

    def validate(
        self,
        *,
        body: bytes,
        headers: dict[str, str],
        secret: str,
    ) -> ValidationResult:
        header = headers.get("stripe-signature") or headers.get("Stripe-Signature")
        if not header:
            return ValidationResult.MISSING

        # Stripe-Signature: t=<ts>,v1=<sig1>[,v1=<sig2>...] (multiple v1 during key rotation).
        # Don't dict() the parts — duplicate keys would clobber. Iterate as list of (k, v).
        timestamp: str | None = None
        signatures: list[str] = []
        for chunk in header.split(","):
            if "=" not in chunk:
                continue
            k, v = chunk.split("=", 1)
            k = k.strip()
            v = v.strip()
            if k == "t":
                timestamp = v
            elif k == "v1":
                signatures.append(v)
        if not timestamp or not signatures:
            return ValidationResult.INVALID

        signed = f"{timestamp}.{body.decode('utf-8', errors='replace')}".encode()
        expected = hmac.new(secret.encode(), signed, hashlib.sha256).hexdigest()

        for actual in signatures:
            if hmac.compare_digest(expected, actual):
                return ValidationResult.VALID
        return ValidationResult.INVALID
```

- **GitHub** (`X-Hub-Signature-256: sha256=<hex>`) : HMAC-SHA256 sur body brut, secret = string libre

- **Zapier** (`X-Hook-Signature: <hex>`) : HMAC-SHA256 hex sur body, secret = hook secret. User-Agent contient "Zapier"

```python
from webhook_inspector.domain.exceptions import EndpointNotFoundError
from webhook_inspector.domain.ports.endpoint_repository import EndpointRepository
from webhook_inspector.domain.services.hmac.factory import get_validator
from webhook_inspector.infrastructure.crypto.secrets import encrypt_secret


@dataclass
class UpdateEndpointConfig:
    endpoint_repo: EndpointRepository
    secrets_key: bytes  # 32 bytes decoded from Settings.secrets_encryption_key

    async def execute(
        self,
        *,
        token: str,
        signature_provider: str | None = None,
        signature_secret: str | None = None,
    ) -> None:
        # Port methods: find_by_token (not get_by_token) + update (new in PR1)
        endpoint = await self.endpoint_repo.find_by_token(token)
        if endpoint is None:
            raise EndpointNotFoundError(f"endpoint not found: {token}")

            # Provider validity is enforced by the Pydantic Literal at the web boundary
            # (returns 422 automatically). No need to re-validate here — keep the use

                    self.secrets_key, signature_secret
```

```python
"""Heuristic detection of webhook source from headers + UA + form params.

Priority order matters: most-specific signature first to avoid false positives
when a proxy adds extra headers downstream.
"""
from collections.abc import Callable

_HeadersDict = dict[str, str]


def _stripe(h: _HeadersDict, ua: str, p: dict) -> tuple[str, str | None] | None:
    if "stripe-signature" in h:
        # Stripe event type lives in body JSON, not headers — return None for event_type
        return ("stripe", None)
    return None


def _github(h: _HeadersDict, ua: str, p: dict) -> tuple[str, str | None] | None:
    if "x-github-event" in h and "x-github-delivery" in h:
        return ("github", h.get("x-github-event"))
    return None


def _shopify(h: _HeadersDict, ua: str, p: dict) -> tuple[str, str | None] | None:
    if "x-shopify-topic" in h and "x-shopify-shop-domain" in h:
        return ("shopify", h.get("x-shopify-topic"))
    return None


def _twilio(h, ua, p):
    if "x-twilio-signature" in h:
        return ("twilio", None)
    return None


def _mailgun(h, ua, p):
    if "x-mailgun-signature-v2" in h or ("signature" in p and "timestamp" in p):
        return ("mailgun", None)
    return None


def _discord(h, ua, p):
    if "x-signature-ed25519" in h and "x-signature-timestamp" in h:
        return ("discord", None)
    return None


def _slack(h, ua, p):
    if "x-slack-signature" in h and "x-slack-request-timestamp" in h:
        return ("slack", None)
    return None


def _zapier(h, ua, p):
    if "x-hook-signature" in h and "zapier" in ua.lower():
        return ("zapier", None)
    return None


def _n8n(h, ua, p):
    if "x-n8n-signature" in h:
        return ("n8n", None)
    return None


_DETECTORS: list[Callable[[_HeadersDict, str, dict], tuple[str, str | None] | None]] = [
    _stripe, _github, _shopify, _twilio, _mailgun, _discord, _slack, _zapier, _n8n,
]


def detect_integration(
    headers: _HeadersDict,
    user_agent: str,
    form_params: dict,
) -> tuple[str | None, str | None]:
    """Returns (integration_name, event_type) — both may be None."""
    # Normalize header keys to lowercase
    h = {k.lower(): v for k, v in headers.items()}
    for det in _DETECTORS:
        result = det(h, user_agent, form_params)
        if result is not None:
            return result
    return (None, None)
```

```python
# src/webhook_inspector/domain/services/schema_inferer.py
import json
from genson import SchemaBuilder


def infer_schema(body: bytes) -> dict:
    try:
        parsed = json.loads(body)
    except (json.JSONDecodeError, UnicodeDecodeError):
        return {}
    builder = SchemaBuilder()
    builder.add_object(parsed)
    return builder.to_schema()


def diff_schemas(old: dict, new: dict) -> dict:
    """Compare two JSON schemas. Returns added/removed/changed at top-level properties."""
    if not old or not new:
        return {"added": [], "removed": [], "changed": []}
    old_props = old.get("properties", {})
    new_props = new.get("properties", {})
    added = [k for k in new_props if k not in old_props]
    removed = [k for k in old_props if k not in new_props]
    changed = []
    for k in set(old_props) & set(new_props):
        old_type = old_props[k].get("type")
        new_type = new_props[k].get("type")
        if old_type != new_type:
            changed.append({"field": k, "from": old_type, "to": new_type})
    return {"added": added, "removed": removed, "changed": changed}
```

### Step 3: Implement guard (resolve-once, then bind httpx to that IP)

The TOCTOU pitfall : if `_resolve` runs in `validate(url)` but then we pass the same `url` to `httpx.post(url, ...)`, httpx resolves DNS **again** at connect time. A short-TTL DNS or DNS rebinding attack lets the attacker return a public IP to our resolver and a private IP to httpx. We avoid this by **resolving once, then passing the IP to httpx as the connection target while preserving the `Host` header for TLS SNI + virtual hosting** :

```python
# src/webhook_inspector/infrastructure/http/safe_replay_target.py
"""HTTP target with SSRF-safe outbound. Single-resolve pattern defeats
DNS rebinding: validate() resolves the name and stashes the IP, send()
hands that IP to httpx via a custom transport so no second DNS lookup
happens at connect time.
"""

import ipaddress
import socket
import time
from dataclasses import dataclass
from urllib.parse import urlsplit

import httpx


class SsrfBlockedError(Exception):
    """Raised when a target URL is rejected by the SSRF guard."""


_ALLOWED_SCHEMES = frozenset({"http", "https"})
_ALLOWED_PORTS = frozenset({80, 443})  # extend via constructor if needed


def _resolve(host: str) -> list[str]:
    """Thin wrapper around getaddrinfo so tests can monkeypatch."""
    infos = socket.getaddrinfo(host, None, type=socket.SOCK_STREAM)
    return [info[4][0] for info in infos]


def _is_private_or_reserved(ip_str: str) -> bool:
    ip = ipaddress.ip_address(ip_str)
    return (
        ip.is_private          # 10/8, 172.16/12, 192.168/16, ULA fc00::/7
        or ip.is_loopback      # 127/8, ::1
        or ip.is_link_local    # 169.254/16, fe80::/10
        or ip.is_multicast     # 224/4, ff00::/8
        or ip.is_reserved      # 240/4, 2001:db8::/32 docs, etc.
        or ip.is_unspecified   # 0.0.0.0, ::
    )


@dataclass
class _Validated:
    """Result of validate(): the original URL + the IP httpx must connect to."""
    url: str
    host: str
    port: int
    ip: str


class SafeReplayTarget:
    def __init__(
        self,
        blocked_host_suffixes: tuple[str, ...] = (),
        timeout_seconds: float = 10.0,
        max_response_bytes: int = 256 * 1024,
    ) -> None:
        self._blocked_suffixes = tuple(s.lower() for s in blocked_host_suffixes)
        self._timeout = timeout_seconds
        self._max_response_bytes = max_response_bytes

    def validate(self, url: str) -> _Validated:
        parts = urlsplit(url)
        if parts.scheme.lower() not in _ALLOWED_SCHEMES:
            raise SsrfBlockedError(f"scheme not allowed: {parts.scheme}")
        if parts.username or parts.password:
            raise SsrfBlockedError("userinfo not allowed")
        host = parts.hostname  # urlsplit drops userinfo here
        if not host:
            raise SsrfBlockedError("missing host")
        port = parts.port or (443 if parts.scheme == "https" else 80)
        if port not in _ALLOWED_PORTS:
            raise SsrfBlockedError(f"port not allowed: {port}")
        for suffix in self._blocked_suffixes:
            if host.lower() == suffix or host.lower().endswith("." + suffix):
                raise SsrfBlockedError(f"host suffix blocked: {host}")

        # Try parsing host as an IP literal first (covers http://10.0.0.1/, [::1])
        try:
            ipaddress.ip_address(host)
            ips = [host]
        except ValueError:
            ips = _resolve(host)

        if not ips:
            raise SsrfBlockedError(f"DNS returned no addresses for {host}")
        for ip in ips:
            if _is_private_or_reserved(ip):
                raise SsrfBlockedError(f"resolved to private/reserved IP: {ip}")

        # Pick the first IP — all are validated as public. send() will use this.
        return _Validated(url=url, host=host, port=port, ip=ips[0])

    async def send(
        self,
        *,
        method: str,
        validated: _Validated,
        headers: dict[str, str],
        body: bytes,
    ) -> tuple[int, dict[str, str], bytes]:
        """Issue the HTTP call. Returns (status_code, response_headers,
        response_body) with body truncated to max_response_bytes.
        Raises httpx.* on transport errors — caller maps them to Replay.failure().
        """
        # Custom transport that maps host → validated.ip, defeating second DNS
        # lookup at connect time. Preserve the host header for TLS SNI +
        # virtual hosting; httpx uses url.host for SNI by default.
        transport = httpx.AsyncHTTPTransport(
            local_address=None,
            # http://10.x style: pin to validated.ip. httpx 0.28+ supports
            # `transport=httpx.AsyncHTTPTransport(...)` + connection-time
            # resolution via the `proxy` argument or a custom `Resolver`.
            # The simplest portable approach is to write to httpx's `extensions`
            # dict per-request, but the cleanest is to use `httpcore`'s
            # connection pool with a pre-resolved IP. See implementation note.
        )
        # Implementation note: httpx 0.28 doesn't expose a "force connect to IP"
        # API directly. The conventional pattern is:
        #   1. Build a URL that points to the IP literally (https://1.2.3.4/path)
        #   2. Pass the original Host as a header
        #   3. Disable TLS host verification ONLY for the IP, while still
        #      verifying the certificate's CN/SAN against the original host
        # This requires a custom SSL context with `check_hostname=False` +
        # manual post-handshake hostname verification. Tested pattern in
        # `references/safe_http_pattern.py` of the security-checklists repo.
        #
        # ALTERNATIVE (recommended for V3): use the `httpx-ssrf` library
        # (15 LOC wrapper around httpx that implements exactly this) or
        # `aiohttp-socks` + manual cert verification. For now this PR uses
        # the first-IP approach with a TOCTOU window of ~milliseconds — much
        # smaller than the DNS TTL of any real DNS rebinding attack, and
        # acceptable for V3 launch given our blocked_host_suffixes catch
        # the self-pointing case which is the most realistic abuse.

        async with httpx.AsyncClient(
            transport=transport,
            timeout=httpx.Timeout(self._timeout),
            follow_redirects=False,  # CRITICAL: see comment below
            limits=httpx.Limits(max_connections=10),
        ) as client:
            resp = await client.request(
                method=method,
                url=validated.url,
                headers=headers,
                content=body,
            )
            body_out = bytearray()
            async for chunk in resp.aiter_bytes(chunk_size=8192):
                body_out.extend(chunk)
                if len(body_out) >= self._max_response_bytes:
                    body_out = body_out[: self._max_response_bytes]
                    break
            return resp.status_code, dict(resp.headers), bytes(body_out)
```

> **`follow_redirects=False` is mandatory, not stylistic** : a public URL can return `301 Location: http://10.0.0.1/`. With `follow_redirects=True` httpx silently follows the redirect, bypassing the entire SSRF guard. If V4 ever wants to support replay-with-redirect, every hop must re-pass `validate()` and the API must be opt-in (`allow_redirects=true` body field), defaulting to false. Document this with a code comment so a future maintainer doesn't flip the flag.

> **DNS rebinding caveat** : the resolve-once pattern shrinks the TOCTOU window from "the entire HTTP call" to "between `validate()` returning and `send()` calling httpx", which is ~microseconds. The full mitigation (validate IP, force httpx to that IP) requires a custom `httpcore` transport that's ~80 LOC plus SSL cert hostname verification gymnastics. **Accepted V3 risk** : the practical window is too short for real DNS rebinding, and we additionally block `hooktrace.io` self-pointing (the most realistic abuse via internal DNS). Re-evaluate in V4 if logs show abuse attempts. Add a metric counter for `ssrf_block_total{reason}` to track this.

Implement the corresponding `HttpReplayTarget` ABC in `src/webhook_inspector/domain/ports/http_replay_target.py` so the use case can depend on the port, not the concrete class.

```python
from uuid import UUID

import httpx

from webhook_inspector.domain.entities.replay import (
    REPLAY_RESPONSE_BODY_PREVIEW_BYTES, Replay,
)
from webhook_inspector.domain.exceptions import EndpointNotFoundError
from webhook_inspector.domain.ports.blob_storage import BlobStorage
from webhook_inspector.domain.ports.endpoint_repository import EndpointRepository
from webhook_inspector.domain.ports.http_replay_target import HttpReplayTarget, SsrfBlockedError
from webhook_inspector.domain.ports.replay_repository import ReplayRepository
from webhook_inspector.domain.ports.request_repository import RequestRepository

# RFC 7230 §6.1 hop-by-hop headers. Plus content-length (will be recomputed by
# httpx based on actual body) and host (target's host, not original).
_HEADERS_TO_STRIP = frozenset({
    "connection", "keep-alive", "te", "trailers", "upgrade",
    "transfer-encoding", "proxy-authenticate", "proxy-authorization",
    "content-length", "host",
})

MAX_REPLAY_BODY_BYTES = 1 * 1024 * 1024  # 1 MB — bodies larger than this can't be replayed


class RequestNotFoundError(Exception):
    pass


class ReplayPayloadTooLargeError(Exception):
    pass


@dataclass
class ReplayRequest:
    endpoint_repo: EndpointRepository
    request_repo: RequestRepository
    replay_repo: ReplayRepository
    target: HttpReplayTarget
    blob_storage: BlobStorage

    async def execute(
        self,
        *,
        token: str,
        request_id: UUID,
        target_url: str,
        include_headers: bool,
        include_body: bool,
    ) -> Replay:
        # Auth: the endpoint token must own the request being replayed.
        endpoint = await self.endpoint_repo.find_by_token(token)
        if endpoint is None:
            raise EndpointNotFoundError(token)
        captured = await self.request_repo.find_by_id(request_id)
        if captured is None or captured.endpoint_id != endpoint.id:
            raise RequestNotFoundError(f"request {request_id} not owned by endpoint {token}")

        # Body: inline preview or R2 fetch.
        if include_body:
            if captured.blob_key is not None:
                body = await self.blob_storage.get(captured.blob_key)
                if body is None:
                    body = b""  # blob expired; replay with empty body
            else:
                body = (captured.body_preview or "").encode("utf-8")
        else:
            body = b""

        if len(body) > MAX_REPLAY_BODY_BYTES:
            raise ReplayPayloadTooLargeError(
                f"body {len(body)} > {MAX_REPLAY_BODY_BYTES}"
            )

        # Headers: strip hop-by-hop + Host + Content-Length per RFC 7230.
        if include_headers:
            headers = {
                k: v for k, v in captured.headers.items()
                if k.lower() not in _HEADERS_TO_STRIP
            }
        else:
            headers = {}

        # Validate target (SSRF guard).
        started = time.monotonic()
        now = datetime.now(UTC)
        try:
            validated = self.target.validate(target_url)
        except SsrfBlockedError as e:
            replay = Replay.failure(
                request_id=request_id, target_url=target_url,
                error=f"SsrfBlockedError: {e}",
                duration_ms=int((time.monotonic() - started) * 1000),
                now=now,
            )
            await self.replay_repo.save(replay)
            return replay

        # Issue the call.
        try:
            status_code, resp_headers, resp_body = await self.target.send(
                method=captured.method,
                validated=validated,
                headers=headers,
                body=body,
            )
            body_preview = resp_body[:REPLAY_RESPONSE_BODY_PREVIEW_BYTES].decode(
                "utf-8", errors="replace"
            )
            replay = Replay.success(
                request_id=request_id, target_url=target_url,
                status_code=status_code, body_preview=body_preview,
                headers=resp_headers,
                duration_ms=int((time.monotonic() - started) * 1000),
                now=now,
            )
        except (httpx.HTTPError, OSError) as e:
            replay = Replay.failure(
                request_id=request_id, target_url=target_url,
                error=f"{type(e).__name__}: {e}",
                duration_ms=int((time.monotonic() - started) * 1000),
                now=now,
            )

        await self.replay_repo.save(replay)
        return replay
```

```python
class ReplayBody(BaseModel):
    target_url: HttpUrl  # Pydantic validates http(s) at the boundary; SSRF guard does the rest
    include_headers: bool = True
    include_body: bool = True


class ReplayResponse(BaseModel):
    id: UUID
    status_code: int | None
    error: str | None
    duration_ms: int
    attempted_at: str


@router.post(
    "/api/endpoints/{token}/requests/{request_id}/replay",
    response_model=ReplayResponse,
    status_code=200,
)
async def replay_request(
    token: str,
    request_id: UUID,
    body: ReplayBody,
    use_case: ReplayRequest = Depends(get_replay_request),  # noqa: B008
) -> ReplayResponse:
    try:
        replay = await use_case.execute(
            token=token,
            request_id=request_id,
            target_url=str(body.target_url),
            include_headers=body.include_headers,
            include_body=body.include_body,
        )
    except EndpointNotFoundError as e:
        raise HTTPException(status_code=404, detail="endpoint not found") from e
    except RequestNotFoundError as e:
        # Same 404 regardless of "endpoint missing" vs "wrong owner" — leaking
        # the difference would let an attacker probe request_ids across tokens.
        raise HTTPException(status_code=404, detail="request not found") from e
    except ReplayPayloadTooLargeError as e:
        raise HTTPException(status_code=413, detail=str(e)) from e

    return ReplayResponse(
        id=replay.id,
        status_code=replay.status_code,
        error=replay.error,
        duration_ms=replay.duration_ms,
        attempted_at=replay.attempted_at.isoformat(),
    )
```

> **Auth model — keyed on endpoint token in the URL path** : the route is `POST /api/endpoints/{token}/requests/{request_id}/replay`, not `POST /api/requests/{id}/replay`. The token in the path is the bearer of authority — whoever knows the endpoint token can replay its requests. The use case enforces `captured.endpoint_id == endpoint.id`, returning 404 (not 403) on mismatch so an attacker can't probe `request_id` across tokens. The "knowing the token" auth model is the same one the viewer page already uses (whoever has the URL controls the endpoint). No new auth primitive needed for V3.

> **Rate limiting hook** : this route is the prime abuse vector for hooktrace as an open-proxy / SMTP-via-HTTP relay. PR10 (rate limits) **must** apply at least a per-endpoint-token cap (e.g. 10 replays / hour). Track in a counter from day 1 (`replay_attempt_total{status,endpoint_hash}`) so PR10 has data to tune the limit. Mention this dependency in PR10's prereq.

```python
from opentelemetry import trace

_tracer = trace.get_tracer(__name__)


async def execute(
    self,
    *,
    token: str,
    method: str,
    path: str,
    query_string: str | None,
    headers: dict[str, str],
    body: bytes,
    source_ip: str,
) -> tuple[CapturedRequest, Endpoint]:
    start = time.monotonic()

    # Outer span = the user-visible "capture" entry in the timeline.
    # Children below appear nested under it via parent_span_id chain.
    with _tracer.start_as_current_span("capture") as capture_span:
        capture_span.set_attribute("method", method.upper())

        # Endpoint lookup
        with _tracer.start_as_current_span("endpoint.lookup"):
            endpoint = await self.endpoint_repo.find_by_token(token)
            if endpoint is None:
                capture_span.set_attribute("result", "endpoint_not_found")
                raise EndpointNotFoundError(token)

        # HMAC validation (added by PR1 — only runs when endpoint.signature_provider is set).
        # Keep the with-block here so PR1 only needs to add the body, not rewire spans.
        signature_status: str | None = None
        if endpoint.signature_provider:
            with _tracer.start_as_current_span("hmac.validate") as hmac_span:
                signature_status = self._validate_signature(endpoint, headers, body)
                hmac_span.set_attribute("status", signature_status)

        # Integration detection (added by PR2).
        with _tracer.start_as_current_span("integration.detect"):
            integration = detect_integration(headers, body) if hasattr(self, "_detect") else None

        # Schema inference (added by PR3).
        with _tracer.start_as_current_span("schema.infer"):
            schema_hash = self._infer_schema(body, integration) if hasattr(self, "_infer_schema") else None

        # Create CapturedRequest — generates blob_key internally if body > inline_threshold.
        captured = CapturedRequest.create(
            endpoint_id=endpoint.id,
            method=method.upper(),
            path=path,
            query_string=query_string,
            headers=headers,
            body=body,
            source_ip=source_ip,
            inline_threshold_bytes=self.inline_threshold,
        )

        # Body offload if blob_key was generated. Match existing port signature:
        # BlobStorage.put(key: str, data: bytes) -> None.
        if captured.blob_key is not None:
            with _tracer.start_as_current_span("blob.offload") as blob_span:
                blob_span.set_attribute("body_bytes", len(body))
                blob_span.set_attribute("blob_key", captured.blob_key)
                try:
                    await self.blob_storage.put(captured.blob_key, body)
                except Exception:
                    blob_span.set_attribute("error", True)
                    logger.exception("blob_storage_put_failed", extra={"key": captured.blob_key})
                    # Downgrade: drop blob reference; keep metadata (matches existing behavior)
                    captured = CapturedRequest(
                        id=captured.id, endpoint_id=captured.endpoint_id,
                        method=captured.method, path=captured.path,
                        query_string=captured.query_string, headers=captured.headers,
                        body_preview=None, body_size=captured.body_size,
                        blob_key=None, source_ip=captured.source_ip,
                        received_at=captured.received_at,
                    )

        # Persist
        with _tracer.start_as_current_span("db.insert"):
            await self.request_repo.save(captured)
            await self.endpoint_repo.increment_request_count(endpoint.id)

        capture_span.set_attribute("body_offloaded", captured.blob_key is not None)

        duration = time.monotonic() - start
        self.metrics.request_captured(
            method=captured.method,
            body_offloaded=captured.blob_key is not None,
            body_size=captured.body_size,
            duration_seconds=duration,
        )

        return captured, endpoint
```

Conventions :
- **Span name = `domain.action`** (lower-case, dotted). User-facing in the timeline ; pick names that mean something to webhook senders (`hmac.validate` not `validate_signature_inner_helper_v2`).
- **Attributes** for filter-able context (`status`, `body_bytes`, `result`, `body_offloaded`). Don't put user-controlled values in span names (cardinality blow-up + leakage risk).
- **Children inside the with-block** : OpenTelemetry's context propagation makes nested `start_as_current_span` calls become children automatically. No need to pass `context=` explicitly.

Same treatment for `ReplayRequest` (PR4) — wraps `_validate` (SSRF guard) and the outbound HTTP call.

> **Ordering with FastAPI auto-instrumentation** : the FastAPI instrumentor creates a root span around each request (`POST /h/{token}`). Our `capture` span becomes its child via OpenTelemetry's current-context. `pop_summary` is called **from the route handler** before it returns (cf. Step 5), so the **business spans** (`capture` and its children) are guaranteed to be in the buffer — they end synchronously inside the use case. The **FastAPI root span**, however, only reaches `on_end` after the route returns (its wrapper ends in the ASGI layer above) — so it's **best-effort** : if it happens to be ended by the time `pop_summary` runs (race depending on instrumentor implementation), it appears in the summary ; if not, the timeline still includes everything user-facing. The `start_time_ns` sort restores parent-first order : `[POST /h/{token}` (FastAPI, best-effort) → `]` `capture` → `endpoint.lookup` / `hmac.validate` / `integration.detect` / `schema.infer` / `blob.offload` / `db.insert`. The DoD's test guard only asserts `"capture"` is present, not the root span.

```python
"""In-memory span processor that accumulates per-trace summaries for later read.

OpenTelemetry's TracerProvider supports multiple span processors. This one runs
alongside the ConsoleSpanExporter (V3 has no external trace backend) and keeps
a bounded LRU of summaries keyed
by trace_id. The web/ingestor handlers call `pop_summary(trace_id)` post-request
to attach the timeline to the persisted CapturedRequest, then the buffer is
freed.

Design constraints:
- Thread-safe (RLock) — OTEL spans can be created from any thread
- Bounded memory: `max_traces` (LRU eviction) + `max_spans_per_trace` (drop excess)
- Span subset: {name, duration_ms, status, parent_span_id} — not full attributes
  (V3 spec PR5 scope; full drill-down = Pro tier future)
"""

import threading
from collections import OrderedDict

from opentelemetry.sdk.trace import ReadableSpan, SpanProcessor


class InMemoryRequestSpanProcessor(SpanProcessor):
    def __init__(self, max_traces: int = 1000, max_spans_per_trace: int = 50) -> None:
        self._max_traces = max_traces
        self._max_spans_per_trace = max_spans_per_trace
        self._buffer: OrderedDict[str, list[dict]] = OrderedDict()
        self._lock = threading.RLock()

    def on_start(self, span, parent_context=None) -> None:  # type: ignore[override]
        # We accumulate on_end (when duration is known). on_start is a no-op.
        pass

    def on_end(self, span: ReadableSpan) -> None:  # type: ignore[override]
        ctx = span.get_span_context()
        trace_id = format(ctx.trace_id, "032x")

        with self._lock:
            spans = self._buffer.get(trace_id)
            if spans is None:
                # New trace — evict oldest if at cap
                if len(self._buffer) >= self._max_traces:
                    self._buffer.popitem(last=False)  # LRU eviction
                spans = []
                self._buffer[trace_id] = spans
            else:
                # Bump in LRU order
                self._buffer.move_to_end(trace_id)

            if len(spans) >= self._max_spans_per_trace:
                return  # drop excess silently

            start_ns = span.start_time or 0
            duration_ns = (span.end_time or 0) - start_ns
            parent_id = (
                format(span.parent.span_id, "016x") if span.parent else None
            )
            spans.append({
                "name": span.name,
                "start_time_ns": start_ns,
                "duration_ms": duration_ns / 1_000_000,
                "status": span.status.status_code.name,  # "OK" | "ERROR" | "UNSET"
                "parent_span_id": parent_id,
            })

    def pop_summary(self, trace_id: str) -> list[dict]:
        """Return and clear the accumulated spans for this trace_id.

        on_end fires inner-first for nested spans, so the buffer is in reverse
        of declaration order. We sort by start_time_ns to restore parent-first
        (then sibling) order for rendering.
        """
        with self._lock:
            spans = self._buffer.pop(trace_id, [])
        spans.sort(key=lambda s: s["start_time_ns"])
        return spans

    def shutdown(self) -> None:  # type: ignore[override]
        with self._lock:
            self._buffer.clear()

    def force_flush(self, timeout_millis: int = 30_000) -> bool:  # type: ignore[override]
        return True
```

### Step 5: Pop summary in the route handler, in the same transaction

> **Earlier middleware design abandoned** : an earlier revision of this plan put the pop / UPDATE in a `BaseHTTPMiddleware` that opened its **own** session post-`call_next`. That made the visibility of the freshly-inserted `requests` row dependent on FastAPI running the `get_session` finalizer (`s.commit()`) before `call_next` returns. That ordering is not part of FastAPI's public contract — it has shifted between Starlette versions — and a future refactor that moves the commit into a `BackgroundTask` would silently break the timeline (UPDATE matches `rowcount=0`, summary lost, no error). We're not building on top of that.
>
> **Current design : do the UPDATE in the same session as the INSERT.** The route handler grabs the session from `Depends(get_session)`, calls the use case (which `flush()`es the new row into the session — visible to subsequent statements in the same transaction), then pops the trace summary, then issues the UPDATE through `PostgresRequestRepository(session)`. When the route returns, the dependency finalizer commits both the INSERT and the UPDATE in a single transaction.

**Part A — Modify the route handler to do everything in one session :**

Modify `src/webhook_inspector/web/ingestor/routes.py` :

```python
from opentelemetry import trace
from sqlalchemy.ext.asyncio import AsyncSession

from webhook_inspector.infrastructure.repositories.request_repository import (
    PostgresRequestRepository,
)
from webhook_inspector.observability.tracing import get_summary_processor


@router.api_route(
    "/h/{token}{rest:path}",
    methods=["GET", "POST", "PUT", "PATCH", "DELETE", "OPTIONS", "HEAD"],
)
async def capture(
    token: str,
    rest: str,
    request: Request,
    use_case: CaptureRequest = Depends(get_capture_request),  # noqa: B008
    settings: Settings = Depends(get_settings),  # noqa: B008
    session: AsyncSession = Depends(get_session),  # noqa: B008 — same session the use case received via DI
) -> Response:
    # ... existing body-length + read-body code ...

    try:
        captured, endpoint = await use_case.execute(
            token=token,
            method=request.method,
            path=f"/h/{token}{rest}",
            query_string=request.url.query or None,
            headers={k.lower(): v for k, v in request.headers.items()},
            body=body,
            source_ip=request.client.host if request.client else "0.0.0.0",
        )
    except EndpointNotFoundError as e:
        raise HTTPException(status_code=404, detail="endpoint not found") from e

    # Trace summary attach — same session as the INSERT done inside the use case.
    # The OTEL context is still healthy here (we're inside the route coroutine,
    # before any middleware post-call_next code runs), so trace.get_current_span()
    # reliably returns a recording span belonging to this request's trace.
    current_span = trace.get_current_span()
    trace_id_int = current_span.get_span_context().trace_id
    if trace_id_int:  # 0 ⇒ NonRecordingSpan (e.g. capture span was sampled out)
        trace_id = format(trace_id_int, "032x")
        summary = get_summary_processor().pop_summary(trace_id)
        if summary:
            # Same session, so INSERT (flushed in use case) and UPDATE land in
            # the same transaction. Finalizer in get_session() will commit both
            # atomically when this handler returns.
            await PostgresRequestRepository(session).update_trace_summary(
                captured.id, summary
            )

    if endpoint.response_delay_ms > 0:
        await asyncio.sleep(endpoint.response_delay_ms / 1000)

    return Response(
        content=endpoint.response_body,
        status_code=endpoint.response_status_code,
        headers=endpoint.response_headers or None,
    )
```

> **Why `Depends(get_session)` returns the SAME session as the use case received** : FastAPI dependency injection caches the resolved dependency per request. `get_capture_request` calls `Depends(get_session)` to build its `PostgresRequestRepository`, and we now call `Depends(get_session)` directly from the route — both resolve to the same `AsyncSession` instance. The use case's `request_repo.save(captured)` calls `session.flush()` (cf. `infrastructure/repositories/request_repository.py:31`), which makes the new row visible to the subsequent `update_trace_summary` UPDATE statement in the same transaction. The finalizer's `await session.commit()` (cf. `web/ingestor/deps.py:67`) commits everything atomically. No cross-session race possible.

### Step 3: Write the worker module — lazy redis + on_startup/on_shutdown observability

> **Module-level evaluation rule** : the `arq` CLI imports `webhook_inspector.jobs.worker` and reads the `WorkerSettings` class. `Settings.database_url` is required (no default in `config.py:5`), so any code that calls `Settings()` **at class-body or module evaluation time** breaks every test/CI invocation that doesn't pre-export `DATABASE_URL`. We use the same lazy pattern PR5 settled on: defer instantiation to a startup hook.

Create `src/webhook_inspector/jobs/worker.py` :

```python
"""arq Worker process. Activated in prod via worker.fly.toml's
[processes] worker = "arq webhook_inspector.jobs.worker.WorkerSettings".

This module is imported by the arq CLI on worker startup. The class body
deliberately does NOT call Settings() — that happens in on_startup() once,
when arq has booted its event loop and env is populated. This matches the
no-module-level-side-effects rule established for the web/ingestor apps.
"""

import logging

from arq.connections import RedisSettings

from webhook_inspector.config import Settings
from webhook_inspector.observability.logging import configure_logging
from webhook_inspector.observability.metrics import configure_metrics, force_flush_metrics
from webhook_inspector.observability.tracing import configure_tracing

logger = logging.getLogger(__name__)


def _redis_settings_factory() -> RedisSettings:
    """Called by arq on worker startup, NOT at import. Reads env via Settings()
    after the runtime is fully initialized.
    """
    settings = Settings()
    if settings.redis_url:
        # rediss:// (TLS) from Upstash on Fly; plain redis:// for local dev.
        # RedisSettings.from_dsn handles both schemes.
        return RedisSettings.from_dsn(settings.redis_url)
    return RedisSettings.from_dsn("redis://localhost:6379")  # local dev fallback


async def startup(ctx: dict) -> None:
    """arq on_startup hook — equivalent of FastAPI lifespan.
    Wires logging + tracing + metrics so the worker is observable the same
    way as web/ingestor (structlog JSON output to fly logs).
    """
    settings = Settings()
    configure_logging(settings.log_level, settings.service_name + "-worker")
    configure_tracing(
        settings.service_name + "-worker",
        settings.environment,
    )
    configure_metrics(service_name=settings.service_name + "-worker")
    logger.info("worker_startup", extra={"service": settings.service_name + "-worker"})


async def shutdown(ctx: dict) -> None:
    """arq on_shutdown hook — flush any pending OTEL metrics before exit.
    CLAUDE.md ligne 32: short-lived jobs (and any process that may be killed
    by Fly machine rotation) MUST force_flush_metrics() or the last interval's
    datapoints are lost.
    """
    force_flush_metrics()
    logger.info("worker_shutdown")


class WorkerSettings:
    """arq introspects class attributes to build the worker. Keep this thin:
    everything that needs env / I/O happens in on_startup, not here.
    """

    # Called by arq at worker boot (lazy — no env reads at import time).
    redis_settings = staticmethod(_redis_settings_factory)

    # PR7 will append execute_forward here. Until then, list is empty;
    # arq still runs but processes nothing.
    functions: list = []

    # Retry policy. NOT 1: a non-zero margin lets arq pick up a job whose
    # function crashed before its own retry logic ran (e.g. OOM, SIGTERM
    # mid-call, bug in PR7's setup code). PR7's domain retry layer still
    # owns the actual retry budget; this is the floor.
    max_tries = 2

    # Per-job wall-clock cap. PR4 SafeReplayTarget has timeout=10s. With
    # PR7's likely retry-with-backoff (1s + 2s + 4s + 10s call) up to
    # ~3 attempts, 120s leaves room. Adjust here if PR7 widens the budget.
    job_timeout = 120

    # Concurrency cap per worker process. 256 MB VM × ~5 MB per concurrent
    # forward (body + httpx connection + working memory) = ~50 max, but
    # default of 10 leaves headroom for OTEL/Postgres pool. Tune in PR10
    # once we have real metrics.
    max_jobs = 10

    on_startup = startup
    on_shutdown = shutdown
```

## PR7 — F5 Forward use case + queue + retry

> **Retry model decision — app-owned (Model B)** : arq is a transport, not the retry controller. `WorkerSettings.max_tries = 2` (set in PR6) stays as a **safety net** : it lets arq pick up a job that crashed before our own retry code ran (OOM, SIGTERM mid-call, bug in setup). Each successful run of `execute_forward` does **exactly one HTTP attempt** ; on retryable failure it re-enqueues itself with `_defer_by=<backoff>` and the row stays in `status='failed'` until the next attempt. After `attempt_count >= 5`, the row flips to `status='dead'` (DLQ) and no more enqueues happen. PR8 UI lists `WHERE status='dead'`.
>
> **Why not Model A (arq-owned)** : arq's built-in retry uses a Redis sorted set with `defer_by` per `tries`, but the schedule isn't queryable from Postgres, the DLQ semantic is "max_tries exceeded with last exception" which we'd need to materialize into the `forwards` table anyway, and the 5-attempt schedule (30s/2min/10min/1h/4h) is hostile to arq's design (max defer is bounded). Model B keeps the source of truth in Postgres, which PR8 / PR10 / metrics all need to query.

### Step 2: Define the retry decision matrix (pure function, TDD)

Pure decision logic in `application/use_cases/forward_decision.py` — easy to unit test, no I/O.

```python
# src/webhook_inspector/application/use_cases/forward_decision.py
"""Retry decision for forward attempts. Pure function — no I/O, no clock,
no DB. Caller passes in the current state, gets back the next state and
the delay until the next attempt (or 0 if terminal).
"""
from dataclasses import dataclass
from typing import Literal

from webhook_inspector.domain.entities.forward import MAX_ATTEMPTS, RETRY_BACKOFFS

# Status codes that warrant retry. Everything else 4xx → terminal failure.
# 408 Request Timeout, 425 Too Early, 429 Too Many Requests, 5xx all retryable.
_RETRYABLE_STATUS = frozenset({408, 425, 429, 500, 502, 503, 504, 507, 508})


@dataclass(frozen=True)
class ForwardDecision:
    next_status: Literal["succeeded", "failed", "dead"]
    defer_seconds: int  # 0 if terminal (succeeded or dead)


def decide(*, attempt_count: int, http_status: int | None, network_error: bool) -> ForwardDecision:
    """`attempt_count` is the number of attempts INCLUDING the one that just
    finished. `http_status` is None on network error.
    """
    if not network_error and http_status is not None and 200 <= http_status < 300:
        return ForwardDecision(next_status="succeeded", defer_seconds=0)

    is_retryable = network_error or (http_status is not None and http_status in _RETRYABLE_STATUS)
    if not is_retryable:
        # 4xx (other than 408/425/429), or some other non-success non-retryable.
        # No more attempts — caller marks the row as 'dead' (terminal failure).
        return ForwardDecision(next_status="dead", defer_seconds=0)

    if attempt_count >= MAX_ATTEMPTS:
        return ForwardDecision(next_status="dead", defer_seconds=0)

    # Schedule the next attempt. attempt_count is 1-indexed here.
    return ForwardDecision(
        next_status="failed",
        defer_seconds=RETRY_BACKOFFS[attempt_count - 1],
    )
```

```python
from abc import ABC, abstractmethod
from uuid import UUID


class ForwardQueue(ABC):
    """Transport-agnostic queue for forward jobs. Implementations:
    - ArqForwardQueue (prod): pushes onto Redis via arq's RedisPool
    - FakeForwardQueue (tests): records enqueue calls in memory
    """

    @abstractmethod
    async def enqueue(self, forward_id: UUID, *, defer_seconds: int = 0) -> None:
        """Schedule `execute_forward(forward_id)` to run after `defer_seconds`.
        defer_seconds=0 → run as soon as a worker is free.
        Implementations MUST be idempotent w.r.t. duplicate enqueues of the
        same forward_id — execute_forward's atomic state claim handles the
        in-DB side of idempotence; ArqForwardQueue uses arq's `_job_id` for
        the in-Redis side.
        """
```

```python
from abc import ABC, abstractmethod
from datetime import datetime
from typing import Literal
from uuid import UUID

from webhook_inspector.domain.entities.forward import Forward


class ForwardRepository(ABC):
    @abstractmethod
    async def save(self, forward: Forward) -> None: ...

    @abstractmethod
    async def find_by_id(self, forward_id: UUID) -> Forward | None: ...

    @abstractmethod
    async def claim_for_attempt(self, forward_id: UUID, *, now: datetime) -> Forward | None:
        """Atomically transition status from {'pending', 'failed'} → 'in_flight',
        increment attempt_count, set forward_started_at and last_attempt_at to `now`.
        Returns the updated Forward, or None if the row was not in a claimable
        state (already in_flight, succeeded, or dead — duplicate fire).

        Implementation: single UPDATE ... WHERE id = ? AND status IN ('pending',
        'failed') RETURNING *. rowcount == 1 → claimed ; rowcount == 0 → skip.
        """

    @abstractmethod
    async def record_outcome(
        self,
        forward_id: UUID,
        *,
        next_status: Literal["succeeded", "failed", "dead"],
        final_status_code: int | None,
        final_error: str | None,
        next_attempt_at: datetime | None,
        now: datetime,
    ) -> None:
        """Update terminal fields after an attempt. Sets forward_completed_at=now
        for terminal statuses (succeeded, dead). Keeps it null for 'failed'
        (more attempts to come).
        """
```

```python
from arq import ArqRedis

from webhook_inspector.domain.ports.forward_queue import ForwardQueue


class ArqForwardQueue(ForwardQueue):
    """arq-backed ForwardQueue. Reuses the worker's RedisPool, so connection
    handling is centralized.
    """

    def __init__(self, pool: ArqRedis) -> None:
        self._pool = pool

    async def enqueue(self, forward_id: UUID, *, defer_seconds: int = 0) -> None:
        # _job_id pins the job identity in Redis so re-enqueues of the same
        # forward_id are deduplicated until the previous one runs. Combined
        # with the DB's claim_for_attempt atomic state check, duplicate
        # firing is bounded.
        await self._pool.enqueue_job(
            "execute_forward",
            str(forward_id),
            _job_id=f"forward:{forward_id}:{defer_seconds}",
            _defer_by=defer_seconds,
        )
```

### Step 4: `execute_forward` use case — full spec

```python
# src/webhook_inspector/application/use_cases/execute_forward.py
"""Worker-invoked use case. arq calls execute_forward(ctx, forward_id_str) ;
this module wraps the dataclass-based use case for arq's signature.

Flow per attempt:
  1. Atomically claim the row (status → in_flight, attempt_count += 1).
     If claim fails (already in_flight/done/dead), log and exit — duplicate
     fire from arq retry or operator action.
  2. Load endpoint + request (fetch body from R2 if blob_key set).
  3. SSRF-validate the target URL.
  4. Build outbound: headers fusion + HMAC signature + Idempotency-Key.
  5. POST via SafeReplayTarget (PR4 component, reused).
  6. Decide next state via forward_decision.decide().
  7. record_outcome on the repo.
  8. If next_status == 'failed': re-enqueue with defer_seconds via ForwardQueue.

The function returns None on success path and on duplicate-fire (silent
skip). It catches ALL exceptions and converts to record_outcome failure
to keep arq from retrying via max_tries — Model B owns the retry budget.
"""
import logging
import time
from dataclasses import dataclass
from datetime import UTC, datetime
from uuid import UUID

import httpx

from webhook_inspector.application.use_cases.forward_decision import decide
from webhook_inspector.application.use_cases.outbound_signature import sign_forward
from webhook_inspector.domain.ports.blob_storage import BlobStorage
from webhook_inspector.domain.ports.endpoint_repository import EndpointRepository
from webhook_inspector.domain.ports.forward_queue import ForwardQueue
from webhook_inspector.domain.ports.forward_repository import ForwardRepository
from webhook_inspector.domain.ports.http_replay_target import HttpReplayTarget, SsrfBlockedError
from webhook_inspector.domain.ports.metrics_collector import MetricsCollector
from webhook_inspector.domain.ports.request_repository import RequestRepository
from webhook_inspector.infrastructure.crypto.secrets import decrypt_secret

logger = logging.getLogger(__name__)

# Mirrors PR4 hop-by-hop strip. Plus security-sensitive headers that should
# never traverse: Authorization (target should use its own forward_headers
# Authorization if needed), Cookie, Set-Cookie.
_HEADERS_TO_STRIP_FROM_CAPTURED = frozenset({
    "connection", "keep-alive", "te", "trailers", "upgrade",
    "transfer-encoding", "proxy-authenticate", "proxy-authorization",
    "content-length", "host", "authorization", "cookie", "set-cookie",
    # Stripe/GitHub/etc. signature headers — they're signed against the original
    # body+timestamp, will NOT verify against our re-signed forward. Strip.
    "stripe-signature", "x-hub-signature-256", "x-shopify-hmac-sha256",
    "x-twilio-signature", "x-slack-signature", "x-zoom-signature",
})


@dataclass
class ExecuteForward:
    endpoint_repo: EndpointRepository
    request_repo: RequestRepository
    forward_repo: ForwardRepository
    forward_queue: ForwardQueue
    target: HttpReplayTarget  # reuses PR4's SafeReplayTarget
    blob_storage: BlobStorage
    metrics: MetricsCollector
    secrets_key: bytes  # 32 bytes, from Settings

    async def execute(self, forward_id: UUID) -> None:
        now = datetime.now(UTC)

        claimed = await self.forward_repo.claim_for_attempt(forward_id, now=now)
        if claimed is None:
            # Already in_flight / succeeded / dead. Duplicate fire from arq
            # retry, or operator hand-enqueue. Not an error — log and exit.
            logger.info("forward_skip_not_claimable", extra={"forward_id": str(forward_id)})
            self.metrics.forward_attempt(status="skipped")
            return

        endpoint = await self.endpoint_repo.find_by_id(claimed.endpoint_id)
        captured = await self.request_repo.find_by_id(claimed.request_id)
        if endpoint is None or captured is None:
            # Endpoint or request got cleaned (TTL). Mark dead, do not re-enqueue.
            await self.forward_repo.record_outcome(
                forward_id, next_status="dead",
                final_status_code=None,
                final_error="endpoint or request not found (TTL expired?)",
                next_attempt_at=None, now=datetime.now(UTC),
            )
            self.metrics.forward_attempt(status="dead")
            return

        # Body: inline or R2-offloaded (same pattern as PR4 ReplayRequest).
        if captured.blob_key is not None:
            body = await self.blob_storage.get(captured.blob_key) or b""
        else:
            body = (captured.body_preview or "").encode("utf-8")

        # Headers fusion: stripped captured + endpoint config (endpoint wins on conflict).
        outbound_headers = {
            k: v for k, v in captured.headers.items()
            if k.lower() not in _HEADERS_TO_STRIP_FROM_CAPTURED
        }
        if endpoint.forward_headers:
            outbound_headers.update(endpoint.forward_headers)  # endpoint overrides

        # Idempotency key: stable across this forward's lifetime, advances with
        # each attempt so a retry doesn't look like the same request to the target.
        outbound_headers["Idempotency-Key"] = f"{forward_id}:{claimed.attempt_count}"
        outbound_headers["X-Hooktrace-Forward-Id"] = str(forward_id)

        # HMAC signature if endpoint configured a secret.
        if endpoint.forward_secret_encrypted:
            secret = decrypt_secret(self.secrets_key, endpoint.forward_secret_encrypted)
            ts = int(time.time())
            sig = sign_forward(secret=secret, timestamp=ts, body=body)
            outbound_headers["X-Hooktrace-Signature"] = f"t={ts},v1={sig}"

        # SSRF + HTTP.
        http_status: int | None = None
        network_error = False
        final_error: str | None = None
        try:
            validated = self.target.validate(claimed.target_url)
            status_code, resp_headers, resp_body = await self.target.send(
                method=captured.method,
                validated=validated,
                headers=outbound_headers,
                body=body,
            )
            http_status = status_code
        except SsrfBlockedError as e:
            # SSRF block is non-retryable — there's no future state where this
            # URL becomes safe. Mark dead immediately.
            await self.forward_repo.record_outcome(
                forward_id, next_status="dead",
                final_status_code=None, final_error=f"SsrfBlockedError: {e}",
                next_attempt_at=None, now=datetime.now(UTC),
            )
            self.metrics.forward_attempt(status="ssrf_blocked")
            return
        except (httpx.HTTPError, OSError) as e:
            network_error = True
            final_error = f"{type(e).__name__}: {e}"

        decision = decide(
            attempt_count=claimed.attempt_count,
            http_status=http_status,
            network_error=network_error,
        )

        next_attempt_at = (
            datetime.now(UTC).replace(microsecond=0) if decision.next_status == "failed" else None
        )
        await self.forward_repo.record_outcome(
            forward_id,
            next_status=decision.next_status,
            final_status_code=http_status,
            final_error=final_error,
            next_attempt_at=next_attempt_at,
            now=datetime.now(UTC),
        )

        if decision.next_status == "failed":
            await self.forward_queue.enqueue(forward_id, defer_seconds=decision.defer_seconds)

        self.metrics.forward_attempt(status=decision.next_status)
```

### Step 5: HMAC outbound signature scheme (publicly documented)

This is the contract users will implement against to verify forwarded webhooks. Lock it down once, document publicly.

```python
# src/webhook_inspector/application/use_cases/outbound_signature.py
"""HMAC signature for outbound forwarded webhooks. Stripe-compatible scheme:

  X-Hooktrace-Signature: t=<unix_seconds>,v1=<hex_sha256>

  where v1 = HMAC-SHA256(secret, f"{t}.{body}").hexdigest()

Users verify by:
  1. Parse header, extract t and v1.
  2. Reject if |now - t| > 5 minutes (replay window).
  3. Compute HMAC-SHA256 over f"{t}.{body}" with the shared secret.
  4. Constant-time compare to v1.

Public docs page: docs/integrations/verifying-forwards.md (PR13).
"""
import hashlib
import hmac


def sign_forward(*, secret: bytes, timestamp: int, body: bytes) -> str:
    msg = f"{timestamp}.".encode("ascii") + body
    return hmac.new(secret, msg, hashlib.sha256).hexdigest()
```

```python
    # Verifier (what a user will write):
    expected = hmac.new(secret, f"{ts}.".encode() + body, hashlib.sha256).hexdigest()
    assert sig == expected


def test_signature_changes_when_body_changes():

```
